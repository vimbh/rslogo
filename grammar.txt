

<program> ::= <expression>+
            | <statement>+
            | <expression>+ <statement>+
            | <statement>+ <expression>+
            
## Expressions return either float or bool values.
<expression> ::= <num_expression>
               | <bool_expression>


<num_expression> ::= <binary_expression>
                   | <query_expression>
                   | <ident_ref>
                   | <Num>

<bool_expression> ::= <comparison_expression> 
                    | <boolean_expression>
                    | <ident_ref>

## Statements describe some state transformation.
<statement> ::= <make_statement> 
              | <add_assign_statement>
              | <direction_statement> 
              | <if_statement> 
              | <while_statement>
              | <pen_status_update> 
              | <pen_color_update> 
              | <pen_pos_update>
              | <procedure_statement>
              | <procedure_reference_statement>

<binary_expression> ::= <binary_operator> <num_expression> <num_expression>

<comparison_expression> ::= <comparison_operator> <num_expression> <num_expression>

<boolean_expression> ::= <boolean_operator> <bool_expression> <bool_expression>

<query_expression> ::= <query_kind>

<make_statement> ::= "MAKE" <identifier> <expression>

<add_assign_statement> ::= "ADDASSIGN" <identifier> <num_expression>

<direction_statement> ::= <direction> <num_expression>

<if_expression> ::= "IF" <bool_expression> "[" <program> "]"

<while_expression> ::= "WHILE" <expression> "[" <program> "]"

<pen_color_update> ::= "PENCOLOR" <num_expression>

<pen_pos_update> ::= <pen_position> <num_expression>

<procedure_statement> ::= "PROCEDURE" <identifier> <identifier>* <program> "PROCEND"

<procedure_reference_expression> ::= <identifier> <identifier>*

## Terminals

<ident_ref> ::= <Num> 
              | <Bool>

<Bool> ::= True 
         | False

<Num> ::= f32

<pen_status_update> ::= "PENUP" | "PENDOWN"

<direction> ::= "FORWARD" | "BACK" | "RIGHT" | "LEFT"

<binary_operator> ::= "+" | "-" | "*" | "/"

<comparisonerator> ::= "EQ" | "NE" | "LT" | "GT"

<booleanerator> ::= "AND" | "OR"

<pen_position> ::= "SETX" | "SETY" | "SETHEADING" | "TURN"

<query_kind> ::= "XCOR" | "YCOR" | "HEADING" | "COLOR"
