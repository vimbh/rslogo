-- CONTEXT FREE GRAMMAR --

Expr := Term 
      | BinExp
     
BinExp := BinOp (Expr) (Term)

Binop := '+'
       | '-'
       | '*'
       | '/'

Binding := Make (Id) (Expr | IdRef)

Direction := Forward
           | Back
           | Left
           | Right

Setter:= Penup
       | Pendown
       | SetColour
       | Setheading
       | SetX
       | SetY

Id := String

Num := Int

IdRef := ':' Id

Term := Num 
      | Id
